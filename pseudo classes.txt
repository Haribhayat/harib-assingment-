  <style>
    /* Change color of link when hovered */
    a:hover {
      color: red;
      text-decoration: underline;
    }

    /* After link is visited */
    a:visited {
      color: purple;
    }

    /* Highlight input when focused */
    input:focus {
      border: 2px solid blue;
      background-color: #f0f8ff;
    }

    /* First child of list */
    li:first-child {
      color: green;
      font-weight: bold;
    }

    /* Last child of list */
    li:last-child {
      color: orange;
      font-style: italic;
    }

    /* Every even child in list */
    li:nth-child(even) {
      background-color: #f2f2f2;
    }

    /* Button hover effect */
    button:hover {
      background-color: black;
      color: white;
      cursor: pointer;
    }
  </style>
